// The brute-force solution with nested loops that I wrote, which timed out on very long strings in the Amazon online assessment.
function getLongestSubStringOne(s) {
  if (!s || s.length === 1) return 0;
 
  let maxLength = 0;
 
  for (let i = 0; i < s.length; i++) {
    for (let j = i + 1; j < s.length; j++) {
      if (s[j] > s[i]) {
        maxLength = Math.max(maxLength, j - i + 1);
      }
    }
  }
 
  return maxLength;
}

function longestLexSubstring(s) {
  if (!s) {
      return 0;
  }
  
  const n = s.length;
  let maxLength = 0;
  let maxChar = 'a';
  let start = 0;
  
  // Process the string in reverse to keep track of the maximum character seen
  // This allows us to identify potential end points for valid substrings
  for (let i = n - 1; i >= 0; i--) {
      // If we find a character that's smaller than our maxChar,
      // we've found a valid substring
      if (s[i] < maxChar) {
          maxLength = Math.max(maxLength, maxChar.index - i + 1);
      }
      // If this character is larger than our maxChar, update maxChar
      if (s[i] >= maxChar) {
          maxChar = s[i];
          maxChar.index = i;
      }
  }
  
  return maxLength;
}
